if(Если) запись видётья так if(a == 5) или же скокрощение if(a). Если мы записываем условие "Если 'a' равно 5 то тогда мы обязательно пишем два знака равно '==' ".
Запись if(a != 5) обозночает что код будет выполняться если 'a' не будет равно 5 скорощение выглядит так if (!a).
else(Иначе) Пример:
	    int a = 5;

            if(a != 5) {
                System.Console.WriteLine("Numder is 5");
            } else {
                System.Console.WriteLine("Numder is not 5");
            }

else if(Или же) записываеться между if и else можно проводить доп проверки сколько угодно раз но это не обезателная структура пример:
	    int a = 5;

            if(a > 7) {
                System.Console.WriteLine("Numder > 7");
            } else if(a < 5){
                System.Console.WriteLine("Numder is < 5");
            } else {
                System.Console.WriteLine("Numder is unrecognized!");
            }

Операторы else и else if не могут существовать без оператора if. Также if должен быть прописан самым первым.
Если одна строчка кода можно удалять фигурные скопки '{}'. Пример:
	    int a = 5;

            if(a > 7) // Это условие.
                System.Console.WriteLine("Numder > 7"); // Это строчка.
            else if(a < 5)
                System.Console.WriteLine("Numder is < 5");
            else
                System.Console.WriteLine("Numder is unrecognized!");
           
&& (И) запись будет выглядеть так:
else if(a == 5 && b) тоесть это услвие сработает если 'a' будет равно '5' и если 'b' ровно 'чему то там'.
									 		Это можно прописывать сколько угодно раз.	
|| (или) запись будет выглядеть так:	
else if(a == 5 || b) тоесть это условие сработает если хотябы одна часть будет верной то код будет выполняться.

Итоговый код.

using System;

namespace project {

    class Program {

        static void Main() {

            int a = 5;
            bool b = false;
            if(a > 7)
                System.Console.WriteLine("Numder > 7");
            else if(a == 5 || b) {
                System.Console.WriteLine("Numder is 5");
                // if(b)
                    System.Console.WriteLine("User has b");
            } else {
                System.Console.WriteLine("Numder is unrecognized!");
            }

        }

    }

}

default: это как else(Иначе) работает также но только с оператором Switch.
Switch-case используеться для проверки одной переменой но с множеством условий также можно проверить только на равенство.



using System;

namespace project {

    class Program {

        static void Main() {
            
            short user_input = Convert.ToInt16(Console.ReadLine());

            switch(user_input) {
                case 5: 
                    System.Console.WriteLine("Mumber is 5"); 
                    break;
                case 15: 
                    System.Console.WriteLine("Mumber is 15"); 
                    break;
                case 20: 
                    System.Console.WriteLine("Mumber is 20"); 
                    break;
                case 250: 
                    System.Console.WriteLine("Mumber is 250"); 
                    break;
                default:
                    System.Console.WriteLine("Mumber is unrecognized!");
                    break;
            }

        }

    }

}
